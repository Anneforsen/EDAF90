%--- Arrays---------------------------------------------------
\begin{frame}[fragile] \frametitle{Arrays}
\begin{itemize}
  \item variable size and type
  \item \code{myArray = [1, 'two', new Number(3)]}
  \item index must be number
  \item size is managed by JavaScript
  \item reading an undefined index returns \code{undefined}
  \item \code{myArray['per'] = 3} - adds a property to the array object
  \item \code{push(), pop(), slice()}
  \item \code{map(), reduce(), forEach()}
  \item \code{for ... of} - iterates over elements
  \item \code{for ... in} - iterates over enumerable object properties
\end{itemize}
\end{frame}
